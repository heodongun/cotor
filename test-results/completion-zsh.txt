#compdef cotor
_cotor_completions() {
  local -a subcmds
  subcmds=(init list run validate template dash web resume checkpoint stats doctor version completion)
  _arguments "1: :->subcmds"
  case $state in
    subcmds)
      _describe 'command' subcmds
    ;;
  esac
}
_cotor_completions "$@"
